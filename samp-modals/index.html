  • DIALOG_STYLE_INPUT

    
Dialog({
	title: "Hello",
	body: "Hello, i am a box!",
	type: "DIALOG_STYLE_INPUT",
	button: {
		confirmButton: "Aceptar",
		denyButton: "Cancelar"
	},
	responseConfirm: function(a) {
		alert("Usted escribio: " + (a.length > 0 ? a : "Nada!"))
	}
})
